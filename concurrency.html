Multithreading

use std::thread;
static NTHREADS: i32 = 10;

//Here's our main thread
fn main(){
//Make a vector for child threads
let mut children = vec![];
for i in 0..NTHREADS {
// Thread creation
children.push(thread::spawn(move || { println!("Wee! I'm thread # {}", i)
}));
// Simple thread join loop.
for thread in children {
let _ = child.join();
}
}

Mutexes own the data

fn use_lock(mutex: &Mutex>) {
let vec = {
// acquire the lock
let mut guard = lock(mutex);
// attempt to return a borrow of the data
access(&mut guard)
// guard is destroyed here, releasing the lock
};
// attempt to access the data outside of the lock.
vec.push(3);
}
