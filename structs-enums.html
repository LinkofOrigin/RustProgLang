Structs and Enums

enums allow custom data types.

structs allow multiple members within one object.


enum MyType {
Data1,
Data2(Box<u32>),
NULL,
}

struct MyObject {
myNum: i32,
myStr: &str,
}

fn main() {
let myStuff = MyObject { myNum: 3, myStr: "foo" };
let myKind = MyType::Data1;
}
