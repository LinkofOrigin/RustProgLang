Structs and Enums

enums allow custom data types.

structs allow multiple members within one object.


enum MyType {
Data1,
Data2(Box<u32>),
NULL,
}

struct MyObject {
myNum: i32,
myStr: &str,
}

fn main() {
let my_kind = MyType::Data1;
let my_stuff = MyObject { myNum: 3, myStr: "foo" };

println!("my_stuff has - {}, {}",
    my_stuff.my_num, my_stuff.my_str);
println!("my_kind has - {:?}", my_kind);
}
